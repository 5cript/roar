165 void setReadRanges(Ranges const& ranges, std::string_view contentType)
166 {
167 if (ranges.ranges.empty())
168 throw std::invalid_argument("Ranges must not be empty");
169
170 if (ranges.ranges.size() > 1)
171 {
172 std::size_t totalFile = fileSize();
173 sequences_.resize(ranges.ranges.size() + 1);
174
175 if (ranges.ranges.empty())
176 throw std::runtime_error("No ranges specified");
177
178 for (auto const& range : ranges.ranges)
179 {
180 if (range.end < range.start)
181 throw std::invalid_argument("end < start");
182 if (range.end > fileSize())
183 throw std::invalid_argument("range.end > file size");
184 }
185
186 std::transform(
187 ranges.ranges.begin(), ranges.ranges.end(), sequences_.rbegin(), [&, this](auto const& range) {
188 auto seq = Sequence{range.start, range.end, totalFile, contentType, splitter_};
189 totalSize_ += seq.headSection.size() + (range.end - range.start);
190 return seq;
191 });
192 sequences_.rbegin()->headSection =
193 sequences_.rbegin()->headSection.substr(2, sequences_.rbegin()->headSection.size() - 2);
194 *sequences_.begin() = Sequence{splitter_};
195 totalSize_ += splitter_.size();
196 }
197 else
198 {
199 sequences_.resize(1);
200 sequences_.front() = Sequence{ranges.ranges.front().start, ranges.ranges.front().end};
201 totalSize_ = ranges.ranges.front().end - ranges.ranges.front().start;
202 file_.seekg(static_cast<std::streamoff>(ranges.ranges.front().start));
203 }
204 }

213 std::size_t read(char* buf, std::size_t amount)
214 {
215 std::size_t origAmount = amount;
216 auto& current = sequences_.back();
217
218 if (current.headConsumed != current.headSection.size())
219 {
220 auto copied = current.headSection.copy(
221 buf, std::min(amount, current.headSection.size() - current.headConsumed), current.headConsumed);
222 buf += copied;
223 current.headConsumed += copied;
224 consumed_ += copied;
225 amount -= copied;
226 if (amount > 0)
227 file_.seekg(static_cast<std::streamoff>(current.start));
228 }
229 file_.read(buf, static_cast<std::streamoff>(std::min(amount, current.end - current.start)));
230 auto gcount = static_cast<std::uint64_t>(file_.gcount());
231 consumed_ += gcount;
232 current.start += gcount;
233 amount -= gcount;
234 if (current.start == current.end)
235 sequences_.pop_back();
236
237 if (sequences_.empty())
238 return (origAmount - amount); // amount should be 0 at this point.
239 if (amount > 0)
240 return (origAmount - amount) + read(buf + gcount, amount);
241 return origAmount;
242 }
