24 {

59 {
60 namespace http = boost::beast::http;
61
62 // Do now allow unsecure connections if not explicitly allowed
63 if (this->serverIsSecure_ && !session.isSecure() && !this->serveInfo_.routeOptions.allowUnsecure)
64 return session.sendStrictTransportSecurityResponse();

66 const auto fileAndStatus = getFileAndStatus(req.target());

68 // File is found, now ask the library user for permissions:
69 switch (std::invoke(
70 this->serveInfo_.handler, *this->listener_, session, req, fileAndStatus, this->serveInfo_.serveOptions))
71 {
72 case (ServeDecision::Continue):
73 break;
74 case (ServeDecision::Deny):
75 return session.sendStandardResponse(boost::beast::http::status::forbidden);
76 case (ServeDecision::Handled):
77 return;
78 };

81 switch (req.method())
82 {
83 case (http::verb::head):
84 {

86 *this->listener_, this->serveInfo_.serveOptions.allowDownload))
87 return session.sendStandardResponse(http::status::method_not_allowed);
88 return sendHeadResponse(session, req, fileAndStatus);
89 }
90 case (http::verb::options):
91 {
92 return sendOptionsResponse(session, req, fileAndStatus);
93 }
94 case (http::verb::get):
95 {
96 if (!unwrapFlexibleProvider<RequestListenerT, bool>(
97 *this->listener_, this->serveInfo_.serveOptions.allowDownload))
98 return session.sendStandardResponse(http::status::method_not_allowed);
99 break;
100 }
101 case (http::verb::delete_):
102 {
103 if (!unwrapFlexibleProvider<RequestListenerT, bool>(
104 *this->listener_, this->serveInfo_.serveOptions.allowDelete))
105 return session.sendStandardResponse(http::status::method_not_allowed);
106 break;
107 }
108 case (http::verb::put):
109 {
110 if (!unwrapFlexibleProvider<RequestListenerT, bool>(
111 *this->listener_, this->serveInfo_.serveOptions.allowUpload))
112 return session.sendStandardResponse(http::status::method_not_allowed);
113 break;
114 }
115 default:

117 return session.sendStandardResponse(http::status::method_not_allowed);
118 }
119 }
120
121 // Handle not found files and directories:
122 if (fileAndStatus.status.type() == std::filesystem::file_type::none && req.method() != http::verb::put)
123 return session.sendStandardResponse(http::status::not_found);
124
125 handleFileServe(session, req, fileAndStatus);

145 FileAndStatus getFileAndStatus(boost::beast::string_view target) const
146 {
147 auto analyzeFile = [this](boost::beast::string_view target) -> FileAndStatus {
148 const auto absolute = jail_.pathAsIsInJail(std::filesystem::path{std::string{target}});
149 if (absolute)
150 {
151 const auto relative = jail_.relativeToRoot(std::filesystem::path{std::string{target}});
152 if (std::filesystem::exists(*absolute))
153 return {.file = *absolute, .relative = *relative, .status = std::filesystem::status(*absolute)};
154 else
155 return {.file = *absolute, .relative = *relative, .status = std::filesystem::file_status{}};
156 }
157 return {.file = {}, .status = std::filesystem::file_status{}};
158 };
159
160 std::pair<std::filesystem::file_type, std::filesystem::path> fileAndStatus;
161 target.remove_prefix(basePath_.size() + (basePath_.size() == 1 ? 0 : 1));
162 return analyzeFile(target);

165 void handleFileServe(Session& session, EmptyBodyRequest const& req, FileAndStatus const& fileAndStatus) const
166 {
167 namespace http = boost::beast::http;
168 switch (req.method())
169 {
170 case (http::verb::get):
171 {
172 if (fileAndStatus.status.type() == std::filesystem::file_type::directory)
173 {
174 if (unwrapFlexibleProvider<RequestListenerT, bool>(
175 *this->listener_, this->serveInfo_.serveOptions.allowListing))
176 return makeListing(session, req, fileAndStatus);
177 else
178 {
179 return download(
180 session,
181 req,
182 getFileAndStatus((basePath_ / fileAndStatus.relative / "index.html").string()));
183 }
184 }
185 else
186 return download(session, req, fileAndStatus);
187 }
188 case (http::verb::delete_):
189 {
190 std::error_code ec;
191 auto type = fileAndStatus.status.type();
192 if (type == std::filesystem::file_type::regular || type == std::filesystem::file_type::symlink)
193 {
194 std::filesystem::remove(fileAndStatus.file, ec);
195 }
196 else if (fileAndStatus.status.type() == std::filesystem::file_type::directory)
197 {
198 if (unwrapFlexibleProvider<RequestListenerT, bool>(
199 *this->listener_, this->serveInfo_.serveOptions.allowDeleteOfNonEmptyDirectories))
200 std::filesystem::remove_all(fileAndStatus.file, ec);
201 else
202 {
203 if (std::filesystem::is_empty(fileAndStatus.file))
204 std::filesystem::remove(fileAndStatus.file, ec);
205 else
206 return session.sendStandardResponse(http::status::forbidden);
207 }
208 }
209 else
210 {
211 ec = std::make_error_code(std::errc::operation_not_supported);
212 }
213
214 if (ec)
215 return session.sendStandardResponse(http::status::bad_request, ec.message());
216 else
217 return (void)session.send<http::empty_body>(req)->status(http::status::no_content).commit();
218 }
219 case (http::verb::put):
220 {
221 return upload(session, req, fileAndStatus);
222 }
223 // Cannot happen:
224 default:
225 return;
226 }

350 void upload(Session& session, EmptyBodyRequest const& req, FileAndStatus const& fileAndStatus) const
351 {
352 namespace http = boost::beast::http;
353 if (!req.expectsContinue())
354 {
355 return (void)session.send<http::string_body>(req)
356 ->status(http::status::expectation_failed)
357 .setHeader(http::field::connection, "close")
358 .body("Set Expect: 100-continue")
359 .commit();
360 }
361
362 switch (fileAndStatus.status.type())
363 {
364 case (std::filesystem::file_type::none):
365 break;
366 case (std::filesystem::file_type::not_found):
367 break;
368 case (std::filesystem::file_type::regular):
369 {
370 if (unwrapFlexibleProvider<RequestListenerT, bool>(
371 *this->listener_, this->serveInfo_.serveOptions.allowOverwrite))
372 break;
373 }
374 default:
375 {
376 return session.sendStandardResponse(http::status::forbidden);
377 }
378 }
379
380 auto contentLength = req.contentLength();
381 if (!contentLength)
382 return session.sendStandardResponse(http::status::bad_request, "Require Content-Length.");
383
384 // TODO: Range Requests
385 auto body = std::make_shared<http::file_body::value_type>();
386 boost::beast::error_code ec;
387 body->open(fileAndStatus.file.string().c_str(), boost::beast::file_mode::write, ec);
388 if (ec)
389 return session.sendStandardResponse(
390 http::status::internal_server_error, "Cannot open file for writing.");
391
392 session.send<http::empty_body>(req)
393 ->status(http::status::continue_)
394 .commit()
395 .then([session = session.shared_from_this(), req, body = std::move(body), contentLength](bool closed) {
396 if (closed)
397 return;
398
399 session->read<http::file_body>(req, std::move(*body))
400 ->bodyLimit(*contentLength)
401 .commit()
402 .then([](auto& session, auto const& req) {
403 session.sendStandardResponse(http::status::ok);
404 })
405 .fail([session](Error const& e) {
406 session->sendStandardResponse(http::status::internal_server_error, e.toString());
407 });
408 });
409 }

411 void download(Session& session, EmptyBodyRequest const& req, FileAndStatus const& fileAndStatus) const
412 {
413 namespace http = boost::beast::http;
414 if (fileAndStatus.status.type() != std::filesystem::file_type::regular &&
415 fileAndStatus.status.type() != std::filesystem::file_type::symlink)
416 return session.sendStandardResponse(http::status::not_found);
417
418 // TODO: Range Requests
419 http::file_body::value_type body;
420 boost::beast::error_code ec;
421 body.open(fileAndStatus.file.string().c_str(), boost::beast::file_mode::read, ec);
422 if (ec)
423 return session.sendStandardResponse(
424 http::status::internal_server_error, "Cannot open file for reading.");
425
426 auto intermediate = session.send<http::file_body>(req, std::move(body));
427 intermediate->preparePayload();
428 intermediate->enableCors(req, this->serveInfo_.routeOptions.cors);
429 auto contentType = extensionToMime(fileAndStatus.file.extension().string());
430 if (contentType)
431 intermediate->contentType(*contentType);
432 intermediate->commit();
